cmake_minimum_required(VERSION 3.27.7)

project(Snow VERSION 1.0.0 LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

#making variable
set(SPACEWARS_GAME_NAME SpaceWars)
set(SNOW_GAME_ENGINE SnowEngine)
set(SDL_LIB_NAME SDL3)


include(FetchContent)

FetchContent_Declare(
	${SDL_LIB_NAME}
	GIT_REPOSITORY https://github.com/libsdl-org/SDL.git
	GIT_TAG release-3.4.0
	GIT_SHALLOW FALSE)

FetchContent_MakeAvailable(${SDL_LIB_NAME})

#check other cmake
add_subdirectory(SnowEngine)
add_subdirectory(SpaceWars)
